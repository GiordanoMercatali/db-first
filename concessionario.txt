ID: INT UNISIGNED NOTNULL UNIQUE AUTOINCREMENT
BRAND: VARCHAR(13) NOTNULL
MODEL: VARCHAR(55) NOTNULL
COUNTRY: VARCHAR(56) NOTNULL
YEAR: YEAR NOTNULL
CLASS: VARCHAR(40) NOTNULL
MOTOR_FUEL: VARCHAR(40) NOTNULL
ENGINE: VARCHAR(100) NULL
HORSEPOWER: TINYINT NULL
LENGTH: MEDIUMINT NULL
WIDTH: MEDIUMINT NULL
HEIGHT: MEDIUMINT NULL
WEIGHT: DOUBLE(4,2) NULL
WHEELBASE: MEDIUMINT NULL
NUMBER_OF_DOORS: TINYINT NULL
COLOR: VARCHAR(30) NULL
HAS_PARKING_SENSORS: TINYINT NOTNULL DEFAULT(0)
HAS_PARKING_CAMERA: TINYINT NOTNULL DEFAULT(0)
